# INSTALL

```bash
$ termux-setup-storage
$ pkg update && pkg upgrade
$ pkg install git -y && pkg install ffmpeg -y && pkg install bash -y && pkg install nodejs -y
$ git clone https://github.com/alanwildan/selfbot
$ cd selfbot
$ bash install.sh
$ npm audit fix
$ npm start
```
